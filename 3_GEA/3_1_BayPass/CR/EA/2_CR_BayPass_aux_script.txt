# ==========================================
# BayPass AUX Model Runs 
# Run the auxiliary BayPass model using the created covariance matrix

#Set working directory 
cd 3_GEA/3_1_BayPass/CR/EA

for i in 1 2 3 4 5
do
seed=$((1000 + RANDOM % 9999))
echo "$seed"
nohup /home/ojhollan/BAM/workingPops/sites/poolFstat/g_baypass -npop 30 -gfile CR_Baypass -seed $seed -efile CR_env_BP.txt -auxmodel -omegafile cr_core.median_mat_omega.out -outprefix cr_aux.$i -scalecov -nval 5000 -burnin 50000 -nthreads 48 &
done
